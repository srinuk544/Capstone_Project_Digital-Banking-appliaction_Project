@Entity@Table(name = "accounts")public class Account {    @Id    @GeneratedValue(strategy = GenerationType.IDENTITY)    private Long accountId;    @NotNull    private Long userId;    @Enumerated(EnumType.STRING)    private AccountType accountType;    @PositiveOrZero    private BigDecimal balance;    @Enumerated(EnumType.STRING)    private AccountStatus status;    private LocalDateTime createdAt = LocalDateTime.now();    // getters & setters}
